function desencriptar() {
  // Obtenemos el valor del campo de texto con id "texto"
  var textoEncriptado = document.getElementById("texto").value;

  // Reemplazamos todas las ocurrencias de "ufat" por "u"
  // utilizando el método replace() y expresión regular /ufat/g
  var textoDesencriptado = textoEncriptado.replace(/ufat/g, "u");

  // Reemplazamos todas las ocurrencias de "ober" por "o"
  // utilizando el método replace() y expresión regular /ober/g
  textoDesencriptado = textoDesencriptado.replace(/ober/g, "o");

  // Reemplazamos todas las ocurrencias de "ai" por "a"
  // utilizando el método replace() y expresión regular /ai/g
  textoDesencriptado = textoDesencriptado.replace(/ai/g, "a");

  // Reemplazamos todas las ocurrencias de "imes" por "i"
  // utilizando el método replace() y expresión regular /imes/g
  textoDesencriptado = textoDesencriptado.replace(/imes/g, "i");

  // Reemplazamos todas las ocurrencias de "enter" por "e"
  // utilizando el método replace() y expresión regular /enter/g
  textoDesencriptado = textoDesencriptado.replace(/enter/g, "e");

  // Mostramos el texto desencriptado en el elemento de párrafo con id "resultado"
  document.getElementById("resultado").textContent = textoDesencriptado;
}
